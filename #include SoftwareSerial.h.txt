#include <SoftwareSerial.h>
#include <AFMotor.h>
#include <Servo.h>

// 4개 모터 (4WD)
AF_DCMotor motor_LF(1); // Left Front
AF_DCMotor motor_LB(2); // Left Back
AF_DCMotor motor_RF(3); // Right Front
AF_DCMotor motor_RB(4); // Right Back

#define BT_RXD A5
#define BT_TXD A4
SoftwareSerial bluetooth(BT_RXD, BT_TXD);

// ===== 서보모터 설정 =====
#define SERVO_PIN 9
Servo myServo;
int servoAngle = 90;      // 시작 각도 (중앙)
const int servoStep = 5;  // 한번에 움직이는 각도

char rec_data = 's';
int speed_val = 70;

void setLeft(int dir) {
  motor_LF.run(dir);
  motor_LB.run(dir);
}

void setRight(int dir) {
  motor_RF.run(dir);
  motor_RB.run(dir);
}

void stopAll() {
  motor_LF.run(RELEASE);
  motor_LB.run(RELEASE);
  motor_RF.run(RELEASE);
  motor_RB.run(RELEASE);
}

void setup() {
  Serial.begin(9600);
  bluetooth.begin(9600);
  Serial.println("4WD Smart Car Start!");

  motor_LF.setSpeed(speed_val);
  motor_LB.setSpeed(speed_val);
  motor_RF.setSpeed(speed_val);
  motor_RB.setSpeed(speed_val);
  stopAll();

  myServo.attach(SERVO_PIN);
  myServo.write(servoAngle); // 초기 위치(90도)로 세팅
}

void loop() {
  if (bluetooth.available()) {
    rec_data = bluetooth.read();
    Serial.println(rec_data);
  }

  switch (rec_data) {
    case 'g': setLeft(FORWARD); setRight(FORWARD); break;
    case 'b': setLeft(BACKWARD); setRight(BACKWARD); break;
    case 'l': setLeft(FORWARD); setRight(RELEASE); break;
    case 'r': setLeft(RELEASE); setRight(FORWARD); break;
    case 'q': setLeft(BACKWARD); setRight(FORWARD); break;
    case 'w': setLeft(FORWARD); setRight(BACKWARD); break;

    // ===== 서보모터 제어 =====
    case 'a': // 서보 왼쪽으로
      servoAngle -= servoStep;
      if (servoAngle < 0) servoAngle = 0;
      myServo.write(servoAngle);
      rec_data = 's'; // 한번 움직이고 멈춤(연속회전 방지)
      break;

    case 'd': // 서보 오른쪽으로
      servoAngle += servoStep;
      if (servoAngle > 180) servoAngle = 180;
      myServo.write(servoAngle);
      rec_data = 's';
      break;

    case 'c': // 서보 중앙(90도)으로 리셋
      servoAngle = 90;
      myServo.write(servoAngle);
      rec_data = 's';
      break;

    case 's':
    default:
      stopAll();
      break;
  }
}
